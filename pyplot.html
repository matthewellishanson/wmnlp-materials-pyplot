from random import randint

# Generate a sequence of 5 random integers
random_integers = [randint(1, 100) for _ in range(5)]
x = np.arange(len(random_integers))

### Using Matplotlib to create a scatter plot and a line plot ###
# Create a scatter plot
figs, axs = plt.subplots(1, 2, figsize=(12, 6))
axs[0].scatter(x, random_integers, color='blue', label='Random Integers')
axs[0].set_title('Scatter Plot of Random Integers')
axs[0].set_xlabel('Index')
axs[0].set_ylabel('Value')
# Create a line plot
axs[1].plot(x, random_integers, color='red', marker='o', label='Random Integers')
axs[1].set_title('Line Plot of Random Integers')
axs[1].set_xlabel('Index')
axs[1].set_ylabel('Value')
# Show the plots
plt.tight_layout()
